Java选择排序

```java
/**
1. 从第一个数开始，跟后面所有的数挨个比较
2. 只要大，则交换位置
3. 每比较一次，则把小的数放在最前面
4. 第一次外循环之后，最小的数放在了最前面
5. 外循环，需要比较的趟数，共需要l-1趟，从第0号元素开始，
   最后一个开始比较的数是a[l-2]，只需比较一次，与a[l-1]
6. 内循环，每趟的数与它后面的数比较，如果大则交换位置，则较小的数放在了前面开始与下面一个比较；
   如果小则继续与下一个数比较，直至最后
   第一趟找出了最小的值，第二趟找出次小值

   例子：
   int[] a = {2,5,9,1,6,4} a.length = 6
    第一趟排序：
			第一次排序：a[0]与a[1]，2和5比较，2<5，不变，a={2 5 9 1 6 4}
			第二次排序：a[0]与a[2]，2和9比较，2<9，不变，a={2 5 9 1 6 4}
			第三次排序：a[0]与a[3]，2和1比较，2>1，变了，a={1 5 9 2 6 4}
			第四次排序：a[0]与a[4]，2和5比较，1<6，不变，a={1 5 9 2 6 4}
			第五次排序：a[0]与a[5]，2和9比较，1<4，不变，a={1 5 9 2 6 4}
	第二趟排序：
			第一次排序：a[1]与a[2]，5和9比较，5<9，不变，a={1 5 9 2 6 4}
			第二次排序：a[1]与a[3]，5和2比较，5>2，变了，a={1 2 9 5 6 4}
			第三次排序：a[1]与a[4]，2和6比较，2<6，不变，a={1 2 9 5 6 4}
			第四次排序：a[1]与a[5]，2和4比较，2<4，不变，a={1 2 9 5 6 4}
	第三趟排序：
			第一次排序：a[2]与a[3]，9和5比较，9>5，变了，a={1 2 5 9 6 4}
			第二次排序：a[2]与a[4]，5和6比较，5<6，不变，a={1 2 5 9 6 4}
			第三次排序：a[2]与a[5]，5和4比较，5>4，变了，a={1 2 4 9 6 5}
	第四趟排序：
			第一次排序：a[3]与a[4]，9和6比较，9>6，变了，a={1 2 4 6 9 5}
			第二次排序：a[3]与a[5]，6和4比较，6>5，变了，a={1 2 4 5 9 6}
	第五趟排序：
			第一次排序：a[4]与a[5]，9和6比较，9>6，变了，a={1 2 4 5 6 9}
*/

public class SelectionSort{
	public static void main(String[] args){
		int[] a = {10,1,7,54,43,6,13,24,2,5,8,3};
		selectionSort(a);
		printArray(a);
	}

	// 接收一个数组，选择排序之后返回排好序的数组
	public static int[] selectionSort(int[] array){
		int l = array.length;
		for (int i=0; i<l-1; i++) {
			for (int j=i+1; j<l; j++) {
				if (array[i]>array[j]) {
					int m=array[i];
					array[i]=array[j];
					array[j]=m;									
				}				
			}
		}

		return array;
	}

	// 打印数组元素，中间以空格相隔
	public static void printArray(int[] array){
		for (int i=0; i<array.length; i++){
			System.out.print(array[i] + " ");
		}
	}
}
```
