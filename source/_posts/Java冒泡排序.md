Java冒泡排序

```java
/**
1. 从第一个数开始，每相邻的两个数挨个比较
2. 只要大，则交换位置
3. 每比较一次，则把小的数放在前面
4. 第一次外循环之后，最大的数放在了最后面，以此类推。
5. 外循环，需要比较的趟数，共需要l-1趟，从第0号元素开始，
6. 内循环，每次比较都是先从a[0]和a[1]开始。第一趟需要比较l-1次，第二趟需要比较l-2次。
   每满一趟，则下一轮比较次数少一次。所以，所以每趟的比较次数为l-i-1。

   例子：
   int[] a = {2,5,9,1,6,4} a.length = 6
    第一趟排序：
			第一次排序：a[0]与a[1]，2和5比较，2<5，不变，a={2 5 9 1 6 4}
			第二次排序：a[1]与a[2]，5和9比较，5<9，不变，a={2 5 9 1 6 4}
			第三次排序：a[2]与a[3]，9和1比较，9>1，变了，a={2 5 1 9 6 4}
			第四次排序：a[3]与a[4]，9和6比较，9>6，变了，a={2 5 1 6 9 4}
			第五次排序：a[4]与a[5]，9和4比较，9>4，变了，a={2 5 1 6 4 9}
	第二趟排序：
			第一次排序：a[0]与a[1]，2和5比较，2<5，不变，a={2 5 1 6 4 9}
			第二次排序：a[1]与a[2]，5和1比较，5>1，变了，a={2 1 5 6 4 9}
			第三次排序：a[2]与a[3]，5和6比较，5<6，不变，a={2 1 5 6 4 9}
			第四次排序：a[3]与a[4]，6和4比较，6>4，变了，a={2 1 5 4 6 9}
	第三趟排序：
			第一次排序：a[0]与a[1]，2和1比较，2>1，变了，a={1 2 5 4 6 9}
			第二次排序：a[1]与a[2]，2和5比较，2<5，不变，a={1 2 5 4 6 9}
			第三次排序：a[2]与a[3]，5和4比较，5>4，变了，a={1 2 4 5 6 9}
	第四趟排序：
			第一次排序：a[0]与a[1]，1和2比较，1<2，不变，a={1 2 4 5 6 9}
			第二次排序：a[1]与a[2]，2和4比较，2<4，不变，a={1 2 4 5 6 9}
	第五趟排序：
			第一次排序：a[0]与a[1]，1和2比较，1<2，不变，a={1 2 4 5 6 9}
*/

public class BubbleSort{
	public static void main(String[] args){
		int[] a = {10,1,7,54,43,6,13,24,2,5,8,3};
		bubbleSort(a);
		printArray(a);
	}

	// 接收一个数组，冒泡排序之后返回排好序的数组
	public static int[] bubbleSort(int[] array){
		int l = array.length;
		for (int i=0; i<l-1; i++) {
			for (int j=0; j<l-i-1; j++) {
				if (array[j]>array[j+1]) {
					int m=array[j];
					array[j]=array[j+1];
					array[j+1]=m;									
				}				
			}
		}

		return array;
	}

	// 打印数组元素，中间以空格相隔
	public static void printArray(int[] array){
		for (int i=0; i<array.length; i++){
			System.out.print(array[i] + " ");
		}
	}
}
```
